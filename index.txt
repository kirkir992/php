<?php
session_start();

// Konfiguracja połączenia z Active Directory
$ldap_host = "ldap://your-domain-controller"; // Zastąp adresem Twojego kontrolera domeny
$ldap_port = 389;
$ldap_domain = "yourdomain.local"; // Zastąp nazwą Twojej domeny
$ldap_base_dn = "DC=yourdomain,DC=local"; // Zastąp odpowiednim Base DN

// Sprawdzanie, czy użytkownik jest już zalogowany
if (isset($_SESSION['username'])) {
    header("Location: dashboard.php");
    exit();
}

// Obsługa formularza logowania
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $username = $_POST['username'];
    $password = $_POST['password'];

    // Połączenie z Active Directory
    $ldap_conn = ldap_connect($ldap_host, $ldap_port);
    if (!$ldap_conn) {
        $error = "Nie można połączyć się z serwerem LDAP.";
    } else {
        // Ustawienie opcji protokołu
        ldap_set_option($ldap_conn, LDAP_OPT_PROTOCOL_VERSION, 3);
        ldap_set_option($ldap_conn, LDAP_OPT_REFERRALS, 0);

        // Tworzenie pełnego DN użytkownika
        $ldap_user = $username . "@" . $ldap_domain;

        // Próba uwierzytelnienia
        $bind = @ldap_bind($ldap_conn, $ldap_user, $password);

        if ($bind) {
            // Pomyślne uwierzytelnienie
            $_SESSION['username'] = $username;
            ldap_close($ldap_conn);
            header("Location: dashboard.php");
            exit();
        } else {
            $error = "Nieprawidłowa nazwa użytkownika lub hasło.";
            ldap_close($ldap_conn);
        }
    }
}
